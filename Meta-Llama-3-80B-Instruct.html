def simulate(self):
try:
self.L1 = float(self.L1_entry.get())
self.L2 = float(self.L2_entry.get())
self.M1 = float(self.M1_entry.get())
self.M2 = float(self.M2_entry.get())

self.V1 = float(self.V1_entry.get())
self.V2 = float(self.V2_entry.get())
self.DURATION = float(self.DURATION_entry.get())
except ValueError:
messagebox.showerror("Error", "Invalid input. Please enter valid numbers.")
return

self.master.withdraw()

t = np.arange(0, self.DURATION, self.DT)
a1 = np.zeros(len(t))
a2 = np.zeros(len(t))
v1 = np.zeros(len(t))
v2 = np.zeros(len(t))

a1[0] = self.A1
a2[0] = self.A2
v1[0] = self.V1
v2[0] = self.V2

for i in range(1, len(t)):
num1 = -G * (2 * self.M1 + self.M2) * np.sin(a1[i-1])
num2 = -self.M2 * G * np.sin(a1[i-1] - 2 * a2[i-1])
num3 = -2 * np.sin(a1[i-1] - a2[i-1]) * self.M2
num4 = v2[i-1]**2 * self.L2 + v1[i-1]**2 * self.L1 * np.cos(a1[i-1] - a2[i-1])
den = self.L1 * (2 * self.M1 + self.M2 - self.M2 * np.cos(2 * a1[i-1] - 2 * a2[i-1]))
a1_acc = (num1 + num2 + num3 * num4) / den

num1 = 2 * np.sin(a1[i-1] - a2[i-1])
num2 = (v1[i-1]**2 * self.L1 * (self.M1 + self.M2))
num3 = G * (self.M1 + self.M2) * np.cos(a1[i-1])
num4 = v2[i-1]**2 * self.L2 * self.M2 * np.cos(a1[i-1] - a2[i-1])
den = self.L2 * (2 * self.M1 + self.M2 - self.M2 * np.cos(2 * a1[i-1] - 2 * a2[i-1]))
a2_acc = (num1 * (num2 + num3 + num4)) / den

v1[i] = v1[i-1] + a1_acc * self.DT
v2[i] = v2[i-1] + a2_acc * self.DT
a1[i] = a1[i-1] + v1[i] * self.DT
a2[i] = a2[i-1] + v2[i] * self.DT
